        function hienThiLoiChuc() {
            // --- BẠN CÓ THỂ THÊM/SỬA LỜI CHÚC TRONG NGOẶC KÉP Ở ĐÂY ---
            const danhSachLoiChuc = [
                "luôn là đóa hoa xinh đẹp nhất, rạng rỡ nhất!",
                "có một ngày 20/10 ngập tràn quà và hoa!",
                "luôn vui vẻ, tươi trẻ và hạnh phúc không chỉ hôm nay mà còn 364 ngày còn lại.",
                "luôn thành công, xinh đẹp và được mọi người yêu quý!",
                "luôn giữ nụ cười trên môi, vì bạn xứng đáng nhận những điều tốt đẹp nhất.",
                "có một ngày lễ thật ý nghĩa bên gia đình và người thương."
            ];
            // --- KẾT THÚC PHẦN SỬA LỜI CHÚC ---
            let ten = document.getElementById("tenNguoiNhan").value;
            if (ten.trim() === "") { ten = "Bạn"; }
            const loiChucNgauNhien = danhSachLoiChuc[Math.floor(Math.random() * danhSachLoiChuc.length)];
            document.getElementById("loiChuc").innerText = `Chúc ${ten} ${loiChucNgauNhien} ❤️`;
        }
